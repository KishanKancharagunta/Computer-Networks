- r0 = a2 + a3 + a0  modulo 2
- r1 = a3 + a2 + a1  modulo 2
- r2 = a1 + a0 + a3  modulo 2

- s0 = b2 + b1 + b0 + q0 modulo 2
- s1 = b3 + b2 + b1 + q1 modulo 2
- s2 = b1 + b0 + b3 + q2 modulo 2
